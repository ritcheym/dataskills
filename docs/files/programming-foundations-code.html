Looping Over Character Vectors

words <- c("it", "was", "the", "dirty", "end", "of", "winter")

for (this_word in words) {
  n_letters <- nchar(this_word)
  block_word <- toupper(this_word)
  cat(block_word, "has", n_letters, "letters\n")
}

Telegram — loop until STOP:

telegram <- c("All", "is", "well", "here",
              "STOP", "This", "is", "fine")

word <- ""
ind <- 0
while (word != "STOP") {
  ind <- ind + 1
  word <- telegram[ind]
  print(word)
}
[1] "All"
[1] "is"
[1] "well"
[1] "here"
[1] "STOP"

Combining Loops and Branches

days <- c("Monday", "Tuesday", "Wednesday",

  } else {
    cat(today, ": keep going...\n")
  }
}

Grading a whole vector:

scores <- c(95, 72, 58, 83, 67)

for (s in scores) {
  if (s >= 90) {

  } else if (s >= 70) {
    g <- "C"
  } else if (s >= 60) {
    g <- "D"
  } else {
    g <- "F"
  }
  cat("Score:", s, "→ Grade:", g, "\n")

Activity: round_sqrt

round_sqrt <- function(x, digits = 0) {
  result <- sqrt(x)
  result <- round(result, digits = digits)
  return(result)
}

round_sqrt(17)
[1] 4
round_sqrt(17, digits = 2)
[1] 4.12

Putting It All Together

A grading function:

grade <- function(score) {
  if (score >= 90) return("A")
  if (score >= 80) return("B")
  if (score >= 70) return("C")
  if (score >= 60) return("D")
  return("F")
}

scores <- c(95, 72, 58, 83, 67)
for (s in scores) {
  cat("Score:", s, "→ Grade:", grade(s), "\n")
}
Score: 95 → Grade: A 
Score: 72 → Grade: C 
Score: 58 → Grade: F 
Score: 83 → Grade: B 
Score: 67 → Grade: D

Activity: FizzBuzz

fizzbuzz <- function(n) {
  if (n %% 3 == 0 & n %% 5 == 0) {
    cat("FizzBuzz\n")
  } else if (n %% 3 == 0) {
    cat("Fizz\n")
  } else if (n %% 5 == 0) {
    cat("Buzz\n")
  } else {
    cat(n, "\n")
  }
}

for (i in 1:20) {
  fizzbuzz(i)
}
